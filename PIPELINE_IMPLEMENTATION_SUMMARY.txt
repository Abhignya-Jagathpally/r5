================================================================================
USAGE EXAMPLES
================================================================================

1. COMPLETE PIPELINE (one command):
   python scripts/run_preprocessing.py --config configs/data_pipeline.yaml \
       --slides data/raw/*.svs --steps tiling stain_norm dedup embeddings

2. INDIVIDUAL STEPS:
   # Tiling only
   from src.data import WSITiler
   tiler = WSITiler(output_dir="data/tiles")
   tiler.process_slides(["slide.svs"], ["s001"])

   # Stain normalization
   from src.data import StainNormalizer
   normalizer = StainNormalizer(method="macenko")
   normalizer.process_tile_directory("data/tiles", "data/normalized")

   # Deduplication
   from src.data import TileDeduplicator
   dedup = TileDeduplicator(hamming_threshold=8)
   dedup.deduplicate_directory("data/normalized", "data/deduplicated")

   # Embeddings
   from src.data import EmbeddingExtractor, EmbeddingStore
   extractor = EmbeddingExtractor(device="cuda")
   embeddings, files = extractor.extract_from_directory("data/deduplicated")
   store = EmbeddingStore("data/embeddings")
   store.add_slide_embeddings("slide_001", embeddings, coordinates)

3. TRAINING WITH EMBEDDINGS:
   from src.data import PatchDataset
   from torch.utils.data import DataLoader
   
   dataset = PatchDataset("data/embeddings", "data/embeddings/metadata.parquet")
   loader = DataLoader(dataset, batch_size=32, num_workers=4)
   for batch in loader:
       embeddings = batch["embedding"]  # Shape: (32, 2048)
